#include "esp_camera.h"
#include <WiFi.h>
#include <WiFiUdp.h>


const char* ssid = "myphone";
const char* password = "12345";


#define PWDN_GPIO_NUM     32
#define RESET_GPIO_NUM    -1
#define XCLK_GPIO_NUM      0
#define SIOD_GPIO_NUM     26
#define SIOC_GPIO_NUM     27
#define Y9_GPIO_NUM       35
#define Y8_GPIO_NUM       34
#define Y7_GPIO_NUM       39
#define Y6_GPIO_NUM       36
#define Y5_GPIO_NUM       21
#define Y4_GPIO_NUM       19
#define Y3_GPIO_NUM       18
#define Y2_GPIO_NUM        5
#define VSYNC_GPIO_NUM    25
#define HREF_GPIO_NUM     23
#define PCLK_GPIO_NUM     22


#define FLASH_LED_GPIO_NUM 4


#define BUZZER_PIN 13


WiFiUDP udp;
const int udpPort = 82;  


WiFiServer streamServer(81);

bool ledState = false;
unsigned long lastLedChange = 0;
const unsigned long ledTimeout = 30000;  


camera_fb_t *fb = NULL;


void generateTone(int pin, int frequency, int duration) {
    if (frequency == 0) {
        digitalWrite(pin, LOW);
        return;
    }
    
    long period = 1000000 / frequency;  
    long cycles = duration * 1000 / period;
    
    for (long i = 0; i < cycles; i++) {
        digitalWrite(pin, HIGH);
        delayMicroseconds(period / 2);
        digitalWrite(pin, LOW);
        delayMicroseconds(period / 2);
    }
}

void setup() {
    Serial.begin(115200);
    Serial.println("\nESP32-CAM Lightweight Streamer");
    

    pinMode(FLASH_LED_GPIO_NUM, OUTPUT);
    digitalWrite(FLASH_LED_GPIO_NUM, LOW);
    

    pinMode(BUZZER_PIN, OUTPUT);
    digitalWrite(BUZZER_PIN, LOW);
    

    camera_config_t config;
    config.ledc_channel = LEDC_CHANNEL_0;
    config.ledc_timer = LEDC_TIMER_0;
    config.pin_d0 = Y2_GPIO_NUM;
    config.pin_d1 = Y3_GPIO_NUM;
    config.pin_d2 = Y4_GPIO_NUM;
    config.pin_d3 = Y5_GPIO_NUM;
    config.pin_d4 = Y6_GPIO_NUM;
    config.pin_d5 = Y7_GPIO_NUM;
    config.pin_d6 = Y8_GPIO_NUM;
    config.pin_d7 = Y9_GPIO_NUM;
    config.pin_xclk = XCLK_GPIO_NUM;
    config.pin_pclk = PCLK_GPIO_NUM;
    config.pin_vsync = VSYNC_GPIO_NUM;
    config.pin_href = HREF_GPIO_NUM;
    config.pin_sscb_sda = SIOD_GPIO_NUM;
    config.pin_sscb_scl = SIOC_GPIO_NUM;
    config.pin_pwdn = PWDN_GPIO_NUM;
    config.pin_reset = RESET_GPIO_NUM;
    config.xclk_freq_hz = 10000000;  
    config.pixel_format = PIXFORMAT_JPEG;
    config.frame_size = FRAMESIZE_QVGA;  
    config.jpeg_quality = 20;         
    config.fb_count = 1;              
    

    esp_err_t err = esp_camera_init(&config);
    if (err != ESP_OK) {
        Serial.printf("Camera init failed: 0x%x\n", err);
        delay(1000);
        ESP.restart();
        return;
    }
    

    WiFi.begin(ssid, password);
    Serial.print("Connecting to WiFi");
    
    int attempts = 0;
    while (WiFi.status() != WL_CONNECTED && attempts < 30) {
        delay(500);
        Serial.print(".");
        attempts++;
        
        // Blink LED while connecting
        digitalWrite(FLASH_LED_GPIO_NUM, !digitalRead(FLASH_LED_GPIO_NUM));
    }
    
    if (WiFi.status() == WL_CONNECTED) {
        Serial.println("\n WiFi connected");
        Serial.print(" IP Address: ");
        Serial.println(WiFi.localIP());
        
        // Success blink pattern on LED
        for(int i = 0; i < 3; i++) {
            digitalWrite(FLASH_LED_GPIO_NUM, HIGH);
            delay(150);
            digitalWrite(FLASH_LED_GPIO_NUM, LOW);
            delay(150);
        }
 


        for(int i = 0; i < 3; i++) {
 
            for(int freq = 800; freq <= 2000; freq += 100) {
                generateTone(BUZZER_PIN, freq, 20);
            }
            delay(50);
     
            for(int freq = 2000; freq >= 800; freq -= 100) {
                generateTone(BUZZER_PIN, freq, 20);
            }
            delay(50);
        }


    } else {
        Serial.println("\n WiFi connection failed!");
        // Error blink pattern on LED
        while(true) {
            digitalWrite(FLASH_LED_GPIO_NUM, HIGH);
            delay(500);
            digitalWrite(FLASH_LED_GPIO_NUM, LOW);
            delay(500);
        }
    }
    

    udp.begin(udpPort);
    Serial.printf(" UDP LED control on port %d\n", udpPort);
    Serial.println("   Send 'LED_ON' or 'LED_OFF' to control LED");
    
    streamServer.begin();
    Serial.println(" Stream server started on port 81");
    Serial.println("   Stream URL: http://" + WiFi.localIP().toString() + ":81/stream");
    

    digitalWrite(FLASH_LED_GPIO_NUM, LOW);
    Serial.println("\n System Ready! Waiting for connections...");
}

void playPoliceSiren() {
    Serial.println(" Motion detected! Playing police siren...");
    

    for(int i = 0; i < 3; i++) {

        for(int freq = 800; freq <= 2000; freq += 100) {
            generateTone(BUZZER_PIN, freq, 20);
        }
        delay(50);

        for(int freq = 2000; freq >= 800; freq -= 100) {
            generateTone(BUZZER_PIN, freq, 20);
        }
        delay(50);
    }
    

    for(int i = 0; i < 6; i++) {
        digitalWrite(FLASH_LED_GPIO_NUM, HIGH);
        delay(100);
        digitalWrite(FLASH_LED_GPIO_NUM, LOW);
        delay(100);
    }
}


void handleUDP() {

    int packetSize = udp.parsePacket();
    if (packetSize) {
        char packetBuffer[20];
        int len = udp.read(packetBuffer, sizeof(packetBuffer) - 1);
        
        if (len > 0) {
            packetBuffer[len] = 0; 
            
            String command = String(packetBuffer);
            command.trim();
            
            if (command == "LED_ON") {
                digitalWrite(FLASH_LED_GPIO_NUM, HIGH);
                ledState = true;
                lastLedChange = millis();
                Serial.println("ðŸ’¡ LED turned ON (UDP command)");
                

                playPoliceSiren();
                

                udp.beginPacket(udp.remoteIP(), udp.remotePort());
                const char* response = "LED_ON_OK";
                udp.write((const uint8_t*)response, strlen(response));
                udp.endPacket();
            }
            else if (command == "LED_OFF") {
                digitalWrite(FLASH_LED_GPIO_NUM, LOW);
                ledState = false;
                lastLedChange = millis();
                Serial.println("ðŸ’¡ LED turned OFF (UDP command)");
                

                generateTone(BUZZER_PIN, 1000, 50);
                

                udp.beginPacket(udp.remoteIP(), udp.remotePort());
                const char* response = "LED_OFF_OK";
                udp.write((const uint8_t*)response, strlen(response));
                udp.endPacket();
            }
        }
    }
    

    if (ledState && (millis() - lastLedChange > ledTimeout)) {
        digitalWrite(FLASH_LED_GPIO_NUM, LOW);
        ledState = false;
        Serial.println("ðŸ’¡ LED auto-off (safety timeout)");
    }
}

void handleStreamClient(WiFiClient &client) {

    client.println("HTTP/1.1 200 OK");
    client.println("Content-Type: multipart/x-mixed-replace; boundary=frame");
    client.println("Access-Control-Allow-Origin: *");
    client.println();
    
    unsigned long lastFrameTime = 0;
    const unsigned long frameInterval = 100; 
    
    while (client.connected()) {

        handleUDP();
        

        unsigned long currentTime = millis();
        if (currentTime - lastFrameTime < frameInterval) {
            delay(10);
            continue;
        }
        lastFrameTime = currentTime;
        

        fb = esp_camera_fb_get();
        if (!fb) {
            Serial.println("Camera capture failed");
            break;
        }
        

        client.print("--frame\r\n");
        client.print("Content-Type: image/jpeg\r\n");
        client.printf("Content-Length: %d\r\n", fb->len);
        client.print("\r\n");
        

        client.write(fb->buf, fb->len);
        client.print("\r\n");
        

        esp_camera_fb_return(fb);
        fb = NULL;
    }
    
    client.stop();
    Serial.println(" Client disconnected");
}

void loop() {

    handleUDP();
    

    WiFiClient client = streamServer.available();
    if (client) {
        Serial.println(" New client connected");
        

        unsigned long timeout = millis() + 5000;
        while (!client.available() && millis() < timeout) {
            delay(10);
            handleUDP(); 
        }
        
        if (client.available()) {
            String request = client.readStringUntil('\r');
            
           
            if (request.indexOf("GET /stream") >= 0 || request.indexOf("GET / ") >= 0) {
                handleStreamClient(client);
            } else {
      
                client.println("HTTP/1.1 200 OK");
                client.println("Content-Type: text/plain");
                client.println();
                client.println("ESP32-CAM Stream Server");
                client.println("Endpoints:");
                client.println("  /stream - Video stream (MJPEG)");
                client.println("UDP port 82 - LED control");
                client.println("Commands: LED_ON, LED_OFF");

           
                generateTone(BUZZER_PIN, 2000, 150);
                delay(100);
                generateTone(BUZZER_PIN, 2500, 150);

                client.stop();
            }
        } else {
            client.stop();
        }
    }
    

    if (WiFi.status() != WL_CONNECTED) {
        Serial.println(" WiFi disconnected, attempting to reconnect...");
        WiFi.reconnect();
        

        for(int i = 0; i < 5; i++) {
            digitalWrite(FLASH_LED_GPIO_NUM, HIGH);
            delay(100);
            digitalWrite(FLASH_LED_GPIO_NUM, LOW);
            delay(100);
        }
        

        generateTone(BUZZER_PIN, 1000, 100);
        delay(100);
        generateTone(BUZZER_PIN, 800, 100);
        

        int reconnectAttempts = 0;
        while (WiFi.status() != WL_CONNECTED && reconnectAttempts < 20) {
            delay(500);
            reconnectAttempts++;
        }
        
        if (WiFi.status() == WL_CONNECTED) {
            Serial.println(" WiFi reconnected");

            generateTone(BUZZER_PIN, 1500, 100);
            delay(100);
            generateTone(BUZZER_PIN, 2000, 100);
        }
    }
    
   
    delay(10);
}
